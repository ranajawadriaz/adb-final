a. SELECT * FROM CUSTOMER WHERE customerID =1122; (Assume file is not ordered) b. SELECT * FROM CUSTOMER WHERE customerID =1515; (Assume file is ordered on customerID) c. SELECT * FROM CUSTOMER WHERE city= ‘Lhr’; (Assume file is not ordered)
